<script>
    /* Hamburger menu toggle */
    function toggleMenu() {
        const menu = document.getElementById("hamburgerMenu");
        menu.style.display = (menu.style.display === "block") ? "none" : "block";
    }
    document.addEventListener("click", function(event){
        const menu = document.getElementById("hamburgerMenu");
        const btn = document.querySelector(".hamburger-btn");
        if(!menu.contains(event.target) && !btn.contains(event.target)){
            menu.style.display = "none";
        }
    });

    // Badge & challenge system
    const challenges = {
        "Monday": {text: "Read 10 pages", badge: "üìö Book Worm"},
        "Tuesday": {text: "Read a short story", badge: "üìñ Story Lover"},
        "Wednesday": {text: "Try a new genre", badge: "üåü Explorer"},
        "Thursday": {text: "Read before bed", badge: "üåô Night Reader"},
        "Friday": {text: "Read 20 pages", badge: "üìö Advanced Reader"},
        "Saturday": {text: "Review a book", badge: "üìù Critic"},
        "Sunday": {text: "Finish a book", badge: "üèÜ Master Reader"}
    };

    let userBadges = JSON.parse(localStorage.getItem("userBadges") || "[]");

    function getTodayChallenge() {
        const day = new Date().toLocaleDateString('en-US', {weekday: 'long'});
        return challenges[day] || challenges["Monday"];
    }

    function initializeChallenge() {
        const challenge = getTodayChallenge();
        document.getElementById("challenge-text").innerText = challenge.text;
    }

    function markDone() {
        const challenge = getTodayChallenge();
        // Count existing badge copies
        let existing = userBadges.filter(b => b.name === challenge.badge);
        let rank = "bronze";
        if(existing.length >= 3 && existing.length < 8) rank="silver";
        if(existing.length >= 8 && existing.length < 13) rank="gold";
        if(existing.length >=13 && existing.length <18) rank="diamond";
        if(existing.length >=18) rank="platinum";

        const newBadge = {
            name: challenge.badge,
            earnedDate: new Date().toLocaleString(),
            rank: rank
        };
        userBadges.push(newBadge);
        localStorage.setItem("userBadges", JSON.stringify(userBadges));
        loadBadges();
        loadProudWall();
        initializeChallenge();
    }

    function loadBadges() {
        const container = document.getElementById("badges");
        if(userBadges.length===0){
            container.innerHTML="<p>No badges earned yet. Complete challenges to start!</p>";
        } else {
            container.innerHTML = userBadges.map(b=>{
                return `<div class="badge-item ${b.rank}"><span class="badge-emoji">${b.name.split(' ')[0]}</span><div class="badge-name">${b.name}</div></div>`;
            }).join('');
        }
    }

    function loadProudWall(){
        const wall = document.getElementById("proudWall");
        if(userBadges.length===0){
            wall.innerHTML="<p>No badges yet.</p>";
        } else {
            wall.innerHTML = userBadges.map(b=>{
                return `<div class="badge-item ${b.rank}"><span class="badge-emoji">${b.name.split(' ')[0]}</span><div class="badge-name">${b.name} (${b.rank.toUpperCase()})</div></div>`;
            }).join('');
        }
    }

    window.addEventListener('load', function(){
        initializeChallenge();
        loadBadges();
        loadProudWall();
    });
</script>
